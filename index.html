Fibonacci Algorithms

Method 1 : recursion

    
    function fib(n) {
        return n <= 1 ? n : fib(n - 1) + fib(n - 2);
    } 
    
    console.log(fib(20));

Method 2 : dynamic programming bottom-up

    
    function newFib(n) {
        let a = 1;
        let b = 1;
        for (let i = 3; i <= n; i++) {
            let c = a + b;
            a = b;
            b = c;
        }
        return b;
    }

    console.log(newFib(20));
